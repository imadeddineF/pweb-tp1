Exercice : Programme en C

#include <stdio.h>


int main() {

int M[MAX_ROWS][MAX_COLS];

int somme_positifs =0;

int lignes, colonnes;


// Demander à l'utilisateur de spécifier le nombre de lignes et de colonnes

printf ( "Entrez le nombre de lignes de la matrice : " );

scanf ( "%d", &lignes );

printf ( "Entrez le nombre de colonnes de la matrice : " );

scanf ( "%d", &colonnes );


// Demander à l'utilisateur de saisir les éléments de la matrice

printf ( "Entrez les éléments de la matrice : " );

for ( int i =0; i <lignes; i ++ ) {

for ( int j =0; j <colonnes; j ++ ) {

printf ( "M[%d][%d] : " , i , j );

scanf ( "%d", &M[i][j] );

}

}


// Calcul de la somme des éléments positifs

for ( int i =0; i <lignes; i ++ ) {

for ( int j =0; j <colonnes; j ++ ) {

if ( M[i][j] >0 ) {

somme_positifs += M[i][j] ;

}

}

}


// Affichage de la somme des éléments positifs

printf ( "La somme des éléments positifs de la matrice est : " , somme_positifs );


return 0;

}
